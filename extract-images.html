Extract Images from Priesmont.com

Instructions:

  1. Open priesmont.com in a new tab
  2. Open the browser's Developer Console (F12 or Right-click → Inspect → Console)
  3. Copy and paste the JavaScript code below into the console
  4. Press Enter to run it
  5. Copy the image URLs that are displayed
  6. Come back here and paste them below, or use them to update index.html

JavaScript Code to Run on priesmont.com:

// Extract all images from priesmont.com (function() { const images = []; const imgElements = document.querySelectorAll('img'); const bgImages = []; // Get all img tags imgElements.forEach(img => { if (img.src && !img.src.includes('data:')) { images.push({ src: img.src, alt: img.alt || 'No alt text', width: img.naturalWidth || img.width, height: img.naturalHeight || img.height }); } }); // Get background images from inline styles document.querySelectorAll('[style*="background-image"]').forEach(el => { const style = el.getAttribute('style'); const match = style.match(/url\(['"]?([^'"]+)['"]?\)/); if (match && match[1] && !match[1].includes('data:')) { bgImages.push(match[1]); } }); // Get background images from computed styles const allElements = document.querySelectorAll('*'); allElements.forEach(el => { const bgImage = window.getComputedStyle(el).backgroundImage; if (bgImage && bgImage !== 'none') { const match = bgImage.match(/url\(['"]?([^'"]+)['"]?\)/); if (match && match[1] && !match[1].includes('data:') && !bgImages.includes(match[1])) { bgImages.push(match[1]); } } }); // Combine and filter const allImages = [ ...images.map(img => img.src), ...bgImages ].filter((url, index, self) => url && url.startsWith('http') && self.indexOf(url) === index && !url.includes('logo') && !url.includes('icon') ); // Display results console.log('=== FOUND IMAGES ==='); allImages.forEach((url, index) => { console.log(`${index + 1}. ${url}`); }); // Create a visual display const resultsDiv = document.createElement('div'); resultsDiv.id = 'extracted-images'; resultsDiv.style.cssText = 'position:fixed;top:0;left:0;right:0;bottom:0;background:rgba(0,0,0,0.9);z-index:99999;overflow-y:auto;padding:20px;'; resultsDiv.innerHTML = `

Found ${allImages.length} Images

${allImages.map((url, index) => `

Image ${index + 1}

${url}
`).join('')}
`; document.body.appendChild(resultsDiv); return allImages; })();

Alternative: Manual Method

If the script doesn't work, you can manually find images:

  1. Right-click on any image on priesmont.com
  2. Select "Copy image address" or "Copy image URL"
  3. Paste the URL into the appropriate place in index.html
